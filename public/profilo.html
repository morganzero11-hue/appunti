// Funzione per caricare i dati all'apertura async function caricaProfilo() { const utenteId = getCookie('utente_id'); // La tua funzione helper per i cookie if (!utenteId) { window.location.href = 'login.html'; return; } try { const res = await fetch(`/api/utente?id=${utenteId}`); const user = await res.json(); if (res.ok) { document.getElementById('nomeIn').value = user.nome || ''; document.getElementById('cognomeIn').value = user.cognome || ''; document.getElementById('emailIn').value = user.email || ''; // Se hai l'immagine if (user.foto_profilo_url) { document.getElementById('avatarImg').src = user.foto_profilo_url; document.getElementById('avatarImg').style.display = 'block'; document.getElementById('avatarPh').style.display = 'none'; } } } catch (err) { console.error("Errore caricamento dati:", err); } } // Funzione per salvare i dati nel DB async function salvaProfilo() { const dati = { id: getCookie('utente_id'), nome: document.getElementById('nomeIn').value, cognome: document.getElementById('cognomeIn').value, email: document.getElementById('emailIn').value }; const res = await fetch('/api/aggiorna-profilo', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(dati) }); if (res.ok) { alert("Profilo aggiornato con successo nel database!"); } } // Avvio document.addEventListener('DOMContentLoaded', caricaProfilo);
